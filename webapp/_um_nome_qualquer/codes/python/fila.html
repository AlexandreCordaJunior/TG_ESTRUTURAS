class No:
    def __init__(self, valor, proximo):
        self.valor = valor
        self.proximo = proximo


class Fila:
    def __init__(self):
        self.cabeca = None
        self.cauda = None

    def enfileirar(self, valor):
        novo = No(valor, None)

        if self.cabeca == self.cauda and self.cabeca == None:
            self.cabeca = self.cauda = novo
        else:
            self.cauda.proximo = novo
            self.cauda = novo
