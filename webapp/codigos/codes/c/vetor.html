#include <stdio.h>
#define TAMANHO 10

void inserir(int *vetor, int valor, int indice) {
    if(indice < TAMANHO) {
        vetor[indice] = valor;
    }
}

void printAll(int *vetor) {
    for(int i = 0; i < TAMANHO; i++) {
        printf("%d ", vetor[i]);
    }
    printf("\n");
}

int main() {
    int vetor[TAMANHO];
    //limpar o vetor
    for(int i = 0; i < TAMANHO; i++) {
        vetor[i] = 0;
    }
    inserir(vetor, 15, 1);
    printAll(vetor);
    inserir(vetor, 16, 0);
    printAll(vetor);
    inserir(vetor, 151, 1);
    printAll(vetor);
    inserir(vetor, 152, 0);
    printAll(vetor);
    inserir(vetor, 153, 1);
    printAll(vetor);
    inserir(vetor, 154, 1);
    printAll(vetor);
    inserir(vetor, 155, 1);
    printAll(vetor);
}
