class Vetor:
    def __init__(self):
        self.TAMANHO = 10
        self.vetor = [None] * 10

    def inserir(self, valor, indice):
        if indice < self.TAMANHO:
            self.vetor[indice] = valor

    def printAll(self):
        for i in range(self.TAMANHO):
            print(self.vetor[i], end=" ")
        print()


vetor = Vetor()
vetor.inserir(15, 1)
vetor.printAll()
vetor.inserir(16, 0)
vetor.printAll()
vetor.inserir(151, 1)
vetor.printAll()
vetor.inserir(152, 0)
vetor.printAll()
vetor.inserir(153, 1)
vetor.printAll()
vetor.inserir(154, 1)
vetor.printAll()
vetor.inserir(155, 1)
vetor.printAll()
