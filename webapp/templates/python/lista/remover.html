def remover(self, indice):
    if(self.topo == None):
        print("NÃ£o pode retirar um elemento de uma lista vazia")
        return

    if self.tamanho < indice:
        print("Erro ao tentar retirar um elemento inexistente de uma lista");
        return

    anterior = None
    atual = self.topo
    for i in range(indice):
        anterior = atual
        atual = atual.proximo

    temporario = atual
    valor = temporario.valor

    if anterior == None:
        self.topo = self.topo.proximo
    else:
        anterior.proximo = atual.proximo
