def inserir(self, valor, indice):
    if indice > self.tamanho:
        print("Não foi possível inserir, pois a lista tem " + str(self.tamanho) + " elementos")
        return

    novo = No(valor, None)
    anterior = None
    atual = self.topo

    for i in range(indice):
        anterior = atual
        atual = atual.proximo

    if anterior == None:
        novo.proximo = atual
        self.topo = novo
    else:
        novo.proximo = atual
        anterior.proximo = novo
    self.tamanho += 1
