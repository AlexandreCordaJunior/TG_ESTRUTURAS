função inserir(fila, valor) {
    novoNo = novo No()
    novoNo.valor = valor
    novoNo.proximo = NULO

    se fila.cabeca == fila.cauda e fila.cabeca == NULO {
        fila.cabeca = novoNo
        fila.cauda = novoNo
    }
    se não {
        fila.cauda.proximo = novoNo
        fila.cauda = novoNo
    }
}
